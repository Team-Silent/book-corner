 CREATE TABLE Books(
   Book_Id varchar2(10),
   Title varchar2(50),
   Author varchar2(30),
   Purchese_price number(10,3),
   Selling_Price number(10,3),
   Stock int,
   PRIMARY KEY (Book_Id)
   );

CREATE TABLE Customers(
    Name varchar2(30),
    Contact_Number varchar2(11),
    Address varchar2(50),
    PRIMARY KEY (Contact_Number)
    );

 CREATE TABLE Book_customer_junction(
    Book_Id varchar2(10),
    Customer_id varchar2(11),
    Quantity int,
    PRIMARY KEY (Book_Id,Customer_id),
    FOREIGN KEY(Book_Id) REFERENCES Books(Book_Id),
    FOREIGN KEY (Customer_id) REFERENCES Customers(Contact_Number)
    );
